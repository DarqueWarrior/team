function CommandVerb-VSTeamCommandNoun {
   [CmdletBinding()]
   param(
      [Parameter(Mandatory = $true, Position = 0)]
      [string] $Param1
   )

   Process {
      
      $body = @{
         queryParam = $Param1
      }
      $bodyJson = $body | ConvertTo-Json -Depth 50 -Compress

      #TODO: add your own code to call the API, make sure you use the correct Api version 
      $resp = _callAPI -Area 'area' -Resource 'resource'  `
         -Method "Post" `
         -ContentType 'application/json' `
         -Version $(_getApiVersion 'Core') `
         -Body $bodyJson

      #TODO: return only data when needed. E.g. remove cmdlets don't usually need to return anything
      return $resp
     
   }
}
